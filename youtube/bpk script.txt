

  <script>
    // Seleciona a div com as classes fornecidas
    const targetDiv = document.querySelector('.pt-1.p-1.overflow-y-auto');

    // HTML que será inserido
    const htmlToInsert = `
        <div class="mb-1 z-2 rounded-lg flex items-start space-x-2">
            <span tabindex="-1" class="flex relative justify-center items-center box-border overflow-hidden align-middle z-0 outline-none data-[focus-visible=true]:z-10 data-[focus-visible=true]:outline-2 data-[focus-visible=true]:outline-focus data-[focus-visible=true]:outline-offset-2 w-8 h-8 text-tiny bg-default text-default-foreground rounded-full bg-gradient-to-br from-[#FFB457] to-[#FF705B]" style="width: 24px; height: 24px; font-size: 10px;">
                <span aria-label="A" class="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 font-normal text-center text-inherit" role="img">A</span>
            </span>
            <div class="flex flex-col">
                <div class="font-bold" style="font-size: 11px;">
                    <span class="font-bold text-zinc-500 dark:text-white">Ana</span>
                    <span class="font-normal ml-2">Opa</span>
                </div>
            </div>
        </div>
    `;

    // Função que adiciona o HTML e faz o scroll para o final
    function addHTML() {
        const newElement = document.createElement('div');
        newElement.innerHTML = htmlToInsert;
        targetDiv.appendChild(newElement.firstElementChild);

        // Scroll automático para o final da div
        targetDiv.scrollTop = targetDiv.scrollHeight;
    }

    // Executa a função a cada 2 segundos
    setInterval(addHTML, 2000);








     // Seleciona os elementos pelo ID
  const inputField = document.getElementById('654e5cbf-8f81-4819-59a7-18ae76273a58');
  const button = document.getElementById('a373c659-1464-9072-5dab-fccfa1b8c5d7');

  // Função para adicionar o conteúdo na div
  function addToDiv() {
        const userInput = inputField.value.trim(); // Obtém o valor do input

        if (userInput) {
            // Cria e adiciona o novo conteúdo na div
            const newElement = document.createElement('div');
            newElement.innerHTML = `
                <div class="mb-1 z-2 rounded-lg flex items-start space-x-2">
                  <span tabindex="-1" class="flex relative justify-center items-center box-border overflow-hidden align-middle z-0 outline-none data-[focus-visible=true]:z-10 data-[focus-visible=true]:outline-2 data-[focus-visible=true]:outline-focus data-[focus-visible=true]:outline-offset-2 w-8 h-8 text-tiny bg-default text-default-foreground rounded-full bg-gradient-to-br from-[#FFB457] to-[#FF705B]" style="width: 24px; height: 24px; font-size: 10px;">
                      <span aria-label="${userInput.charAt(0).toUpperCase()}" class="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 font-normal text-center text-inherit" role="img">Y</span>
                  </span>
                  <div class="flex flex-col">
                      <div class="font-bold" style="font-size: 11px;">
                          <span class="font-bold text-zinc-500 dark:text-white">You</span>
                          <span class="font-normal ml-2">${userInput}</span>
                      </div>
                  </div>
              </div>
            `;
            targetDiv.appendChild(newElement.firstElementChild);

            // Faz scroll automático para o final
            targetDiv.scrollTop = targetDiv.scrollHeight;

            // Limpa o conteúdo do input
            inputField.value = '';
        }
    }

    // Adiciona evento de clique ao botão
    button.addEventListener('click', addToDiv);

    // Adiciona evento de tecla ao input
    inputField.addEventListener('keydown', (event) => {
        if (event.key === 'Enter') {
            addToDiv();
        }
    });

</script>
